```jsx
// CreateTodo.jsx
// Environment: Browser

// This doesn't actually import `todo.telefunc.js`:
// Telefunc transforms `todo.telefunc.js` into a
// thin HTTP client.
import { onNewTodo } from './todo.telefunc.js'

async function onClick(form) {
  const text = form.input.value
  // Behind the scenes, Telefunc makes an HTTP request.
  await onNewTodo(text)
}

function CreateTodo() {
  return (
    <form>
      <input input="text"></input>
      <button onClick={onClick}>Add To-Do</button>
    </form>
  )
}
```
